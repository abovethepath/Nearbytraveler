/* Prevent horizontal scroll anywhere */

/* Let flex/grid children shrink text instead of forcing overflow */

/* Map popups / info panes: wrap long lines */
.mapboxgl-popup-content, .gm-style .gm-style-iw, .leaflet-popup-content {
  max-width: calc(100vw - 2rem);
  overflow-wrap: anywhere;
  word-break: break-word;
  white-space: normal;
}
