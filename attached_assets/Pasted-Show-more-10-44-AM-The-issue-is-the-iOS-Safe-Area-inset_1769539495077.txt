Nuclear option - iOS Safe Area fix:

Add this to your index.css:

css
/* iOS Safe Area Fix for Full-Width Elements */
@supports (padding: max(0px)) {
  .bg-gradient-to-r,
  [class*="bg-gradient"] {
    padding-left: max(env(safe-area-inset-left), 0.75rem) !important;
    padding-right: max(env(safe-area-inset-right), 0.75rem) !important;
    width: 100vw !important;
    margin-left: calc(-1 * max(env(safe-area-inset-left), 0px)) !important;
    margin-right: calc(-1 * max(env(safe-area-inset-right), 0px)) !important;
  }
}
