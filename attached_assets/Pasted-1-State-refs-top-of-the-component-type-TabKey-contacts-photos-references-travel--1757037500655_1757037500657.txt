type TabKey = 'contacts' | 'photos' | 'references' | 'travel' | 'countries';

const [activeTab, setActiveTab] = React.useState<TabKey>('contacts');

const tabRefs = {
  contacts: React.useRef<HTMLDivElement>(null),
  photos: React.useRef<HTMLDivElement>(null),
  references: React.useRef<HTMLDivElement>(null),
  travel: React.useRef<HTMLDivElement>(null),
  countries: React.useRef<HTMLDivElement>(null),
};

function openTab(key: TabKey) {
  setActiveTab(key);
  // Wait for the panel to render, then scroll it into view (good for Replit preview)
  requestAnimationFrame(() => {
    tabRefs[key].current?.scrollIntoView({ behavior: 'smooth', block: 'start' });
  });
}

4) (Optional) Persist tab in the URL (nice for reloads)

If you use Wouter:

const [loc, setLoc] = useLocation();
React.useEffect(() => {
  const q = new URLSearchParams(loc.split('?')[1] || '');
  const t = q.get('tab') as TabKey | null;
  if (t) setActiveTab(t);
}, []);

function openTab(key: TabKey) {
  setActiveTab(key);
  const base = loc.split('?')[0];
  setLoc(`${base}?tab=${key}`, { replace: true });
  requestAnimationFrame(() => tabRefs[key].current?.scrollIntoView({ behavior: 'smooth', block: 'start' }));
}


This pattern works great inside Replit’s preview: click a tab → the right panel renders (others are hidden) and the page scrolls that panel into view, so you can always see the content even in a narrow iframe.
