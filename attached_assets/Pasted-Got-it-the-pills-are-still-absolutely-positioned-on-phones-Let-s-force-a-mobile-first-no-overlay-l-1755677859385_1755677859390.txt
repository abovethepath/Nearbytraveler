Got it—the pills are still absolutely-positioned on phones. Let’s force a mobile-first, no-overlay layout and only overlay on md+. Drop this EventCard in (or mirror its structure in your card):

import { MapPin, Calendar as CalendarIcon, Users } from "lucide-react";
import { Button } from "@/components/ui/button";

export function HomeEventCard({ ev }: { ev: any }) {
  return (
    <article className="relative overflow-hidden rounded-2xl bg-gray-800/40 shadow-sm text-left">
      {/* Image */}
      <div className="relative">
        <img
          src={ev.imageUrl}
          alt={ev.title}
          className="w-full aspect-[16/9] object-cover"
          loading="lazy"
        />

        {/* md+ overlay for category only */}
        {ev.category && (
          <>
            <div className="hidden md:block absolute inset-x-0 bottom-0 h-16 bg-gradient-to-t from-black/60 to-transparent" />
            <span className="hidden md:inline-flex absolute top-3 right-3 z-10 rounded-full px-3 py-1 text-xs font-semibold bg-gray-900/80 text-white">
              {ev.category}
            </span>
          </>
        )}
      </div>

      {/* MOBILE pills (NOT overlay) */}
      <div className="md:hidden px-4 pt-3 flex flex-wrap gap-2">
        {ev.category && (
          <span className="inline-flex rounded-full px-3 py-1 text-xs font-semibold bg-gray-900/80 text-white">
            {ev.category}
          </span>
        )}
        {/* put your other pills here: “Local T”, “1 attending”, etc. */}
        {ev.isLocal && (
          <span className="inline-flex rounded-full px-3 py-1 text-xs font-medium bg-gray-200/20 text-white">
            Local T
          </span>
        )}
        {typeof ev.attendingCount === "number" && (
          <span className="inline-flex rounded-full px-3 py-1 text-xs font-medium bg-purple-600 text-white">
            {ev.attendingCount} attending
          </span>
        )}
      </div>

      {/* Content */}
      <div className="p-4 md:p-5">
        <h3 className="text-white text-base md:text-lg font-semibold leading-snug line-clamp-2">
          {ev.title}
        </h3>

        {ev.description && (
          <p className="mt-1 text-sm text-gray-300 leading-relaxed whitespace-normal break-words [overflow-wrap:anywhere]">
            {ev.description}
          </p>
        )}

        {/* Meta: wraps on small screens */}
        <div className="mt-3 flex flex-col gap-2 sm:flex-row sm:flex-wrap sm:items-center">
          {ev.address && (
            <div className="min-w-0 flex items-center gap-2 text-sm text-gray-300">
              <MapPin className="h-4 w-4 shrink-0" />
              <span className="truncate">{ev.address}</span>
            </div>
          )}
          {ev.date && (
            <div className="min-w-0 flex items-center gap-2 text-sm text-gray-300">
              <CalendarIcon className="h-4 w-4 shrink-0" />
              <span className="truncate">{ev.date}</span>
            </div>
          )}
          {typeof ev.attendingCount === "number" && (
            <div className="min-w-0 flex items-center gap-2 text-sm text-gray-300 md:hidden">
              <Users className="h-4 w-4 shrink-0" />
              <span className="truncate">{ev.attendingCount} attending</span>
            </div>
          )}
        </div>

        {/* Actions */}
        <div className="mt-4 flex gap-2">
          <Button className="flex-1 sm:flex-none">Join</Button>
          <Button variant="outline" className="flex-1 sm:flex-none">
            More
          </Button>
        </div>
      </div>
    </article>
  );
}

Why this fixes your screenshot

No absolute pills on mobile → nothing overlaps the title/description/meta.

Category stays overlayed only on md+; on mobile it’s in normal flow.

Meta rows use flex-wrap, min-w-0, and truncate so long addresses/dates don’t collide.

If you must keep overlay pills on mobile

Wrap the image in a container that reserves space for them:

<div className="relative pb-12"> <!-- ~48px reserved -->
  <img className="w-full aspect-[16/9] object-cover" />
  <div className="absolute bottom-2 left-2 right-2 flex gap-2 flex-wrap z-10">
    <!-- pills -->
  </div>
</div>


Also double-check the parent of your cards isn’t adding text-center. If your section wrapper has text-center, move it to just the header and set text-left on the card root.
