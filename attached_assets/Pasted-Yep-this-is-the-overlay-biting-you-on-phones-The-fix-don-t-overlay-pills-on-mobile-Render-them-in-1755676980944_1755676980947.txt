Yep—this is the overlay biting you on phones. The fix: don’t overlay pills on mobile. Render them in normal flow below the image, and only overlay on md+. Also force the card text to be left-aligned and let meta rows wrap.

Here’s a drop-in EventCard that won’t overlap on mobile:

import { MapPin, Calendar as CalendarIcon, Users } from "lucide-react";
import { Button } from "@/components/ui/button";

type Event = {
  id: number;
  title: string;
  description?: string;
  imageUrl?: string;
  category?: string;
  address?: string;
  date?: string;
  attendingCount?: number;
};

export function EventCard({ event }: { event: Event }) {
  return (
    <article className="rounded-2xl bg-gray-800/40 shadow-sm overflow-hidden text-left">
      {/* Image */}
      <div className="relative">
        <img
          src={event.imageUrl}
          alt={event.title}
          className="w-full aspect-[16/9] object-cover"
          loading="lazy"
        />

        {/* md+ overlay chips with scrim */}
        <div className="hidden md:block absolute inset-x-0 bottom-0 h-16 bg-gradient-to-t from-black/60 to-transparent" />
        <div className="hidden md:flex absolute bottom-3 left-3 right-3 z-10 gap-2 flex-wrap">
          {event.category && (
            <span className="rounded-full px-3 py-1 text-xs font-semibold bg-gray-900/80 text-white">
              {event.category}
            </span>
          )}
          {/* put other pills here if you have them */}
        </div>
      </div>

      {/* MOBILE chips (not overlay) */}
      <div className="md:hidden px-4 pt-3 flex flex-wrap gap-2">
        {event.category && (
          <span className="rounded-full px-3 py-1 text-xs font-semibold bg-gray-900/80 text-white">
            {event.category}
          </span>
        )}
        {/* other pills here */}
      </div>

      {/* Content */}
      <div className="p-4 md:p-5">
        <h3 className="text-white text-lg font-semibold leading-snug line-clamp-2">
          {event.title}
        </h3>

        {event.description && (
          <p className="mt-1 text-sm text-gray-300 leading-relaxed break-words">
            {event.description}
          </p>
        )}

        {/* Meta — wraps on small screens */}
        <div className="mt-3 flex flex-col gap-2 sm:flex-row sm:flex-wrap sm:items-center">
          {event.address && (
            <div className="min-w-0 flex items-center gap-2 text-sm text-gray-300">
              <MapPin className="h-4 w-4 shrink-0" />
              <span className="truncate">{event.address}</span>
            </div>
          )}
          {event.date && (
            <div className="min-w-0 flex items-center gap-2 text-sm text-gray-300">
              <CalendarIcon className="h-4 w-4 shrink-0" />
              <span className="truncate">{event.date}</span>
            </div>
          )}
          <div className="min-w-0 flex items-center gap-2 text-sm text-gray-300">
            <Users className="h-4 w-4 shrink-0" />
            <span className="truncate">{event.attendingCount ?? 0} attending</span>
          </div>
        </div>

        {/* Actions */}
        <div className="mt-4 flex gap-2">
          <Button className="flex-1 sm:flex-none">Join</Button>
          <Button variant="outline" className="flex-1 sm:flex-none">
            More
          </Button>
        </div>
      </div>
    </article>
  );
}


If you must keep pills over the image on mobile, reserve space so they don’t sit on top of text:

<div className="relative pb-12"> <!-- reserve ~48px -->
  <img className="w-full aspect-[16/9] object-cover" />
  <div className="absolute bottom-2 left-2 right-2 z-10 flex flex-wrap gap-2">
    <!-- pills -->
  </div>
</div>


Also make sure your existing card/container doesn’t inherit text-center. Set text-left on the card root and meta rows, and keep these helpers on long lines: min-w-0, truncate, flex-wrap, break-words.

Wire this in and the overlap goes away on phones. If you paste your current EventCard JSX, I’ll convert it to this structure line-for-line.
